Inside the chart component's `<script>`, add a helper near where the chart data is prepared:

```ts
import type { LatLon, RoutePoint } from '../routing/types';

/**
 * For each waypoint, find the time on the route polyline at which the route
 * is closest to that waypoint. Returns an array of Unix-ms timestamps.
 */
function waypointEtas(route: RoutePoint[], waypoints: LatLon[]): number[] {
    return waypoints.map(wp => {
        let bestTime = route[0]?.time ?? 0;
        let bestDist = Infinity;
        for (const p of route) {
            const dy = p.lat - wp.lat;
            const dx = (p.lon - wp.lon) * Math.cos(wp.lat * Math.PI / 180);
            const d = dx * dx + dy * dy;
            if (d < bestDist) { bestDist = d; bestTime = p.time; }
        }
        return bestTime;
    });
}

export let waypoints: LatLon[] = [];
$: waypointTimes = selectedRoute ? waypointEtas(selectedRoute.path, waypoints) : [];
```

(`selectedRoute` is the route currently being rendered — adapt the reference to whatever the host component calls it.)

import { Chart } from 'chart.js';

const waypointAnnotationPlugin = {
    id: 'waypointAnnotations',
    afterDatasetsDraw(chart: any, _args: any, opts: any) {
        const ts: number[] = opts?.times ?? [];
        if (!ts.length) return;
        const xScale = chart.scales.x;
        if (!xScale) return;
        const ctx = chart.ctx;
        ctx.save();
        ctx.setLineDash([4, 4]);
        ctx.strokeStyle = 'rgba(96,165,250,0.8)';
        ctx.lineWidth = 1.5;
        for (let i = 0; i < ts.length; i++) {
            const x = xScale.getPixelForValue(ts[i]);
            if (x == null || Number.isNaN(x)) continue;
            ctx.beginPath();
            ctx.moveTo(x, chart.chartArea.top);
            ctx.lineTo(x, chart.chartArea.bottom);
            ctx.stroke();
            // Waypoint badge
            ctx.setLineDash([]);
            ctx.fillStyle = '#3b82f6';
            ctx.beginPath();

            ctx.fillText(String(i + 1), x, chart.chartArea.top + 10);
            ctx.setLineDash([4, 4]);

};
